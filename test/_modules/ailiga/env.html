[docs]def get_env_name(env):
    """Get the name of the env."""
    if callable(env):
        env = env()
    if isinstance(env, gym.Env):
        return env.spec.id
    elif isinstance(env, PettingZooEnv):
        return env.env.metadata["name"]
    else:
        raise Exception("Unknown env type")
